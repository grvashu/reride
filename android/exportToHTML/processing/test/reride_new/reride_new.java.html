411          public void timeView() { 
412              pushStyleMatrix(); 
413              pgl.beginDraw(); 
414              pgl.translate(pgl.width / 2, pgl.height / 2); 
415              pgl.stroke(255); 
416              pgl.strokeWeight(60); 
417              pgl.noFill(); 
418              pgl.colorMode(HSB); 
419              float start = 0; 
420              for (int i = 0; i < data.size(); i++) { 
421                  Data now = data.get(i); 
422                  float tilt = now.tilt(); 
423                  pgl.stroke(map(tilt, MIN_TILT, MAX_TILT, 0, 255), 200, 200, 50); 
424                  pgl.arc(0, 0, 200, 200, start, start + (PI / 120), OPEN); 
425                  start += PI / 120; 
426              } 
427              pgl.colorMode(RGB); 
428              pgl.endDraw(); 
429              popStyleMatrix(); 
430          }

432          public void abstractPosture() { 
433              Data now = data.get(index); 
434              pushStyleMatrix(); 
435   
436              try { 
437                  now.streamData(index); 
438              } catch (Exception e) { 
439              } 
440              pgr.beginDraw(); 
441   
442              pgr.translate(pgr.width / 2, pgr.height / 2); 
443              float tilt = now.tilt(); 
444              pgr.colorMode(HSB); 
445              c = color(map(tilt, MIN_TILT, MAX_TILT, 0, 255), 200, 200); 
446              pgr.fill(c); 
447              pgr.beginShape(); 
448              float new_tilt = +map(tilt, MIN_TILT, MAX_TILT, MIN_DISP_TILT, MAX_DISP_TILT); 
449              //bottom left 
450              pgr.vertex(-pgr.width / 3, (pgr.height / 3) - new_tilt); 
451              //top left 
452              pgr.vertex((-pgr.width / 3) + TRAP_OFFSET, now.lateral_pos()); 
453              //top right 
454              pgr.vertex((pgr.width / 3) - TRAP_OFFSET, now.lateral_pos()); 
455              //bottom right 
456              pgr.vertex(pgr.width / 3, (pgr.height / 3) + new_tilt); 
457              pgr.endShape(); 
458              pgr.strokeWeight(10); 
459              pgr.stroke(255, 0, 0); 
460              pgr.ellipse(0, now.lateral_pos(), map(now.lateral_pos(), -100, 400, 200, 50), map(now.lateral_pos(), -100, 400, 200, 10)); 
461              pgr.colorMode(RGB); 
462              pgr.fill(255); 
463              pgr.textSize(50); 
464              pgr.text("x:" + ax + ",y:" + ay + ",z:" + az, 0, 0); 
465              pgr.endDraw(); 
466   
467              popStyleMatrix(); 
468              delay(500); 
469          }
